program test;
    var x, r: integer;

    integer Fibonacci(integer n);
        begin
            if (n < 3) then
                return 1;

            return Fibonacci(n - 2) + Fibonacci(n - 1)
        end;

    begin
        print("Wpisz liczbę całkowitą: ");
        read(x);
        print("Wynik: ");
        r = Fibonacci(x);
        print(r)
    end.
